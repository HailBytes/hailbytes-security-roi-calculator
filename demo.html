📋 Integration — as simple as this
<!-- 1. Load via CDN -->
<script type="module"
  src="https://cdn.jsdelivr.net/gh/HailBytes/hailbytes-security-roi-calculator@main/hailbytes-roi-calculator.js"></script>

<!-- 2. Use the element -->
<hailbytes-roi-calculator theme="dark"></hailbytes-roi-calculator>

<!-- 3. Handle the result event -->
<script>
  document.querySelector('hailbytes-roi-calculator')
    .addEventListener('roi-calculated', (e) => {
      const { roi, netBenefit, paybackMonths } = e.detail;
      console.log(`ROI: ${roi.toFixed(0)}%`);
      console.log(`Net Benefit: $${netBenefit.toLocaleString()}`);
    });
</script>
